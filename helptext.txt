Usage: dither input [flags]

Input formats: png, jpg, qoi
Output formats: png, qoi

-v
	verbose output
	[default: false]

-y
	luminance only
	[default: false]

-d <no/floyd/2d>
	dither type
	[default: floyd]

--force
	force overwrite
	[default: false]

-<R/G/B/A/Y> <0/1/2/2.6/2.7/3-8>
	bits per channel
	[default: 2.6 2.7 2.6 3 1]

-f <1-3>
	luma conversion formula
	[default: 2]

-o <filename>
	output file
	[default: path/to/input-d.png]

-O <qoi/png>
	changes autogenerated filename extention
	[default: png]

-s <num> <num>
	resize image
	if negative number in one of parameters provided,
	calculate that parameter to keep aspect ratio

-r <CW/CCW>
	rotate image
	[default: false]

-h/--help/-help
	this help
