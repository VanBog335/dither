Usage: dither input [flags]

Input formats: png, jpg, qoi
Output formats: png, qoi

-v
	verbose output
	[default: false]

-y
	luminance only
	[default: false]

-d <no/floyd/2d>
	dither type
	[default: floyd]

--force
	force overwrite
	[default: false]

-<R/G/B/A/Y> <0/1/2/2.6/2.7/3-8>
	bits per channel
	[default: 2.6 2.7 2.6 3 1]

-f <1-3>
	luma conversion formula

-o <filename>
	output file
	[default: path/to/input-d.png]

-s <num> <num> (WIP)
	size of image

-h/--help/-help
	this help
